if echo "$COMMAND" | grep -qE "..."; then

    if [ -f "$marker" ]; then
      # consumer-side self-heal: revalidate marker target before blocking

      echo "BLOCKED: coverage validator 실패 ($marker, target=<X>)." >&2
      echo "  복구: <validator command 예시>" >&2
